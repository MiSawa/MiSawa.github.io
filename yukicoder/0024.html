bool solve(){
    int n;
    cin >> n;
    int A = bit(10)-1;
    rep(_, n){
        int B = 0;
        rep(t, 4){ int x; cin >> x; B |= bit(x); }
        string r; cin >> r;
        if(r == "NO") B = ~B;
        A &= B;
    }
    rep(i, 10) if(A>>i&1) cout << i << endl;
    return true;
}
